Hàm hiển thị và nộp bài --- window.renderQuiz = function() { const quizDiv = document.getElementById('quiz'); if (!quizDiv) return; quizDiv.innerHTML = window.currentQuizData.map((item, i) => `
Câu ${i+1}: ${item.question}
${['a','b','c','d'].map(key => ` `).join('')}
`).join(''); }; window.submitQuiz = function() { let score = 0; let answered = 0; window.currentQuizData.forEach((item, i) => { const selected = document.querySelector(`input[name="q${i}"]:checked`); if (selected) { answered++; if (selected.value === item.correct) score++; } }); if (answered < window.currentQuizData.length && !confirm("Bạn chưa hoàn thành hết các câu hỏi. Vẫn nộp bài?")) return; alert(`Bài làm hoàn thành!\nSố câu đúng: ${score} / ${window.currentQuizData.length}`); document.getElementById('quiz-screen').style.display = 'none'; document.getElementById('start-screen').style.display = 'block'; }; // --- 4. Bảng xếp hạng --- window.showRanking = function() { const script = document.createElement('script'); script.src = `${API_URL}?action=getRanking&callback=handleRankingData`; document.body.appendChild(script); script.onload = () => script.remove(); }; window.handleRankingData = function(data) { document.getElementById('start-screen').style.display = 'none'; document.getElementById('quiz-screen').style.display = 'none'; document.getElementById('result-screen').style.display = 'none'; document.getElementById('rank-board').style.display = 'block'; const tbody = document.getElementById('rank-body'); tbody.innerHTML = data.map(item => `${item.ten}${item.diem}${item.mon}`).join(''); }; // --- 5.
